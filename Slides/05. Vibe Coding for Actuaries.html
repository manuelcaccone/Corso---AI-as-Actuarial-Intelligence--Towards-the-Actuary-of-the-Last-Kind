# Funzioni helper
quick_chat <- function(prompt) {
  if (!exists(".quick_chat_obj")) {
    .quick_chat_obj <<- ellmer::chat_anthropic()
  }
  .quick_chat_obj$chat(prompt)
}

# Funzione per chat veloce con Groq (se disponibile)
quick_groq <- function(prompt, model = "llama-3.1-8b-instant") {
  if (exists("groq_chat")) {
    groq_chat(prompt = prompt, model = model)
  } else {
    message("groqR non √® disponibile. Usa quick_chat() invece.")
  }
}

# Funzione per confronto velocit√†
speed_test <- function(prompt) {
  cat("üöÄ Test di velocit√† per:", prompt, "\n\n")
  
  # Test Groq
  if (exists("groq_chat")) {
    start_time <- Sys.time()
    groq_result <- groq_chat(prompt, model = "llama-3.1-8b-instant")
    groq_time <- as.numeric(Sys.time() - start_time)
    cat("‚ö° Groq:", groq_time, "secondi\n")
  }
  
  # Test ellmer
  start_time <- Sys.time()
  ellmer_result <- quick_chat(prompt)
  ellmer_time <- as.numeric(Sys.time() - start_time)
  cat("üß† ellmer:", ellmer_time, "secondi\n\n")
  
  return(list(groq = groq_result, ellmer = ellmer_result))
}
